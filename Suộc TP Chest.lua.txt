local function transformToSatan()
    -- Change character appearance
    for _, part in pairs(Character:GetDescendants()) do
        if part:IsA("BasePart") and part.Parent:IsA("Model") then
            part.BrickColor = BrickColor.new("Really red")
            if part.Name == "Head" then
                part.BrickColor = BrickColor.new("Maroon")
            end
        end
    end
    
    -- Add horns
    createHorns()
    
    -- Add aura
    createAura()
    
    -- Change face
    local face = Character.Head:FindFirstChild("face")
    if face then
        face.Texture = "rbxasset://textures/face.png"
        face.Color3 = Color3.fromRGB(255, 0, 0)
    end
    
    -- Modify humanoid properties
    Humanoid.WalkSpeed = 25
    Humanoid.JumpPower = 75
    
    -- Show skills
    SkillsFrame.Visible = true
    
    -- Dark atmosphere
    local ColorCorrection = Instance.new("ColorCorrectionEffect")
    ColorCorrection.Name = "SatanEffect"
    ColorCorrection.TintColor = Color3.fromRGB(255, 200, 200)
    ColorCorrection.Saturation = -0.2
    ColorCorrection.Parent = Lighting
end

local function revertTransform()
    -- Reset character appearance
    for _, part in pairs(Character:GetDescendants()) do
        if part:IsA("BasePart") and part.Parent:IsA("Model") then
            part.BrickColor = BrickColor.new("Light orange")
            if part.Name == "Head" then
                part.BrickColor = BrickColor.new("Light orange")
            end
        end
    end
    
    -- Remove horns
    for _, horn in pairs(horns) do
        horn:Destroy()
    end
    horns = {}
    
    -- Remove aura
    if aura then
        aura:Destroy()
        aura = nil
    end
    
    -- Remove wings
    if wings then
        wings:Destroy()
        wings = nil
    end
    
    -- Reset face
    local face = Character.Head:FindFirstChild("face")
    if face then
        face.Texture = "rbxasset://textures/face.png"
        face.Color3 = Color3.fromRGB(255, 255, 255)
    end
    
    -- Reset humanoid properties
    Humanoid.WalkSpeed = 16
    Humanoid.JumpPower = 50
    
    -- Hide skills
    SkillsFrame.Visible = false
    
    -- Remove atmosphere effect
    local effect = Lighting:FindFirstChild("SatanEffect")
    if effect then
        effect:Destroy()
    end
end

-- Skill Functions
local function castHellfire()
    local fireball = Instance.new("Part")
    fireball.Name = "Hellfire"
    fireball.Size = Vector3.new(4, 4, 4)
    fireball.Shape = Enum.PartType.Ball
    fireball.Material = Enum.Material.Neon
    fireball.BrickColor = BrickColor.new("Really red")
    fireball.TopSurface = Enum.SurfaceType.Smooth
    fireball.BottomSurface = Enum.SurfaceType.Smooth
    fireball.CanCollide = false
    fireball.CFrame = RootPart.CFrame + RootPart.CFrame.lookVector * 5
    fireball.Parent = workspace
    
    local fire = Instance.new("Fire")
    fire.Size = 15
    fire.Heat = 25
    fire.Parent = fireball
    
    local bodyVelocity = Instance.new("BodyVelocity")
    bodyVelocity.Velocity = Mouse.Hit.lookVector * 100
    bodyVelocity.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    bodyVelocity.Parent = fireball
    
    fireball.Touched:Connect(function(hit)
        if hit.Parent ~= Character and hit.Parent:FindFirstChild("Humanoid") then
            hit.Parent.Humanoid:TakeDamage(30)
            local explosion = Instance.new("Explosion")
            explosion.Position = fireball.Position
            explosion.BlastRadius = 20
            explosion.BlastPressure = 100000
            explosion.Parent = workspace
            fireball:Destroy()
        end
    end)
    
    Debris:AddItem(fireball, 5)
end

local function shadowTeleport()
    local targetPos = Mouse.Hit.Position + Vector3.new(0, 3, 0)
    
    -- Create shadow effect at start
    local shadow1 = Instance.new("Part")
    shadow1.Size = Vector3.new(5, 0.1, 5)
    shadow1.Position = RootPart.Position - Vector3.new(0, 3, 0)
    shadow1.BrickColor = BrickColor.new("Really black")
    shadow1.Material = Enum.Material.Neon
    shadow1.Transparency = 0.5
    shadow1.Anchored = true
    shadow1.CanCollide = false
    shadow1.Parent = workspace
    
    local shadowTween = TweenService:Create(
        shadow1,
        TweenInfo.new(0.5),
        {Size = Vector3.new(10, 0.1, 10), Transparency = 1}
    )
    shadowTween:Play()
    Debris:AddItem(shadow1, 0.5)
    
    -- Teleport
    RootPart.CFrame = CFrame.new(targetPos)
    
    -- Create shadow effect at end
    local shadow2 = shadow1:Clone()
    shadow2.Position = targetPos - Vector3.new(0, 3, 0)
    shadow2.Size = Vector3.new(10, 0.1, 10)
    shadow2.Transparency = 1
    shadow2.Parent = workspace
    
    local shadowTween2 = TweenService:Create(
        shadow2,
        TweenInfo.new(0.5),
        {Size = Vector3.new(5, 0.1, 5), Transparency = 0.5}
    )
    shadowTween2:Play()
    Debris:AddItem(shadow2, 0.5)
end

local function soulSteal()
    local radius = 30
    local position = RootPart.Position
    
    -- Create soul steal effect
    local sphere = Instance.new("Part")
    sphere.Name = "SoulStealSphere"
    sphere.Shape = Enum.PartType.Ball
    sphere.Size = Vector3.new(1, 1, 1)
    sphere.Position = position
    sphere.BrickColor = BrickColor.new("Royal purple")
    sphere.Material = Enum.Material.ForceField
    sphere.Transparency = 0.7
    sphere.Anchored = true
    sphere.CanCollide = false
    sphere.Parent = workspace
    
    local sphereTween = TweenService:Create(
        sphere,
        TweenInfo.new(1, Enum.EasingStyle.Quad),
        {Size = Vector3.new(radius*2, radius*2, radius*2), Transparency = 1}
    )
    sphereTween:Play()
    Debris:AddItem(sphere, 1)
    
    -- Steal health from nearby players
    for _, target in pairs(workspace:GetDescendants()) do
        if target:FindFirstChild("Humanoid") and target ~= Character then
            local targetHumanoid = target.Humanoid
            local targetRoot = target:FindFirstChild("HumanoidRootPart")
            if targetRoot then
                local distance = (targetRoot.Position - position).Magnitude
                if distance <= radius then
                    local damage = 20
                    targetHumanoid:TakeDamage(damage)
                    Humanoid.Health = math.min(Humanoid.MaxHealth, Humanoid.Health + damage/2)
                    
                    -- Create soul effect
                    local soul = Instance.new("Part")
                    soul.Size = Vector3.new(1, 1, 1)
                    soul.Shape = Enum.PartType.Ball
                    soul.BrickColor = BrickColor.new("Royal purple")
                    soul.Material = Enum.Material.Neon
                    soul.Transparency = 0.5
                    soul.Anchored = true
                    soul.CanCollide = false
                    soul.Position = targetRoot.Position
                    soul.Parent = workspace
                    
                    local soulTween = TweenService:Create(
                        soul,
                        TweenInfo.new(0.5, Enum.EasingStyle.Linear),
                        {Position = RootPart.Position, Size = Vector3.new(0.1, 0.1, 0.1), Transparency = 1}
                    )
                    soulTween:Play()
                    Debris:AddItem(soul, 0.5)
                end
            end
        end
    end
end
